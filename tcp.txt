//server tcp
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>

#define MAX_DATA 1024

void process_data(char *msg); // ðŸ‘ˆ Youâ€™ll define this per question

int main(int argc, char *argv[]) {
    int serSock, cliSock, len, data_len;
    char msg[MAX_DATA];
    struct sockaddr_in server, client;

    if (argc != 2) {
        printf("Usage: %s <port>\n", argv[0]);
        exit(1);
    }

    serSock = socket(AF_INET, SOCK_STREAM, 0);
    if (serSock < 0) {
        perror("Socket");
        exit(1);
    }

    server.sin_family = AF_INET;
    server.sin_port = htons(atoi(argv[1]));
    server.sin_addr.s_addr = INADDR_ANY;
    bzero(&server.sin_zero, 8);

    if (bind(serSock, (struct sockaddr*)&server, sizeof(server)) < 0) {
        perror("Bind");
        exit(1);
    }

    listen(serSock, 10);
    printf("Server listening on port %s...\n", argv[1]);

    len = sizeof(client);

    while (1) {
        cliSock = accept(serSock, (struct sockaddr*)&client, &len);
        if (cliSock < 0) {
            perror("Accept");
            continue;
        }

        printf("Client connected: %s:%d\n", inet_ntoa(client.sin_addr), ntohs(client.sin_port));

        while ((data_len = recv(cliSock, msg, MAX_DATA, 0)) > 0) {
            msg[data_len] = '\0';
            printf("Received: %s", msg);

            process_data(msg); // ðŸ‘ˆ Call your custom function

            send(cliSock, msg, strlen(msg), 0);
            printf("Sent: %s\n", msg);
        }

        close(cliSock);
    }

    close(serSock);
    return 0;
}

//client tcp
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>

#define MAX_DATA 1024

int main(int argc, char *argv[]) {
    int sock, data_len;
    char input[MAX_DATA], output[MAX_DATA];
    struct sockaddr_in server;

    if (argc != 3) {
        printf("Usage: %s <IP> <port>\n", argv[0]);
        exit(1);
    }

    sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock < 0) {
        perror("Socket");
        exit(1);
    }

    server.sin_family = AF_INET;
    server.sin_port = htons(atoi(argv[2]));
    server.sin_addr.s_addr = inet_addr(argv[1]);
    bzero(&server.sin_zero, 8);

    if (connect(sock, (struct sockaddr*)&server, sizeof(server)) < 0) {
        perror("Connect");
        exit(1);
    }

    printf("Connected to server %s:%s\n", argv[1], argv[2]);

    while (1) {
        printf("Enter message: ");
        fgets(input, MAX_DATA, stdin);
        send(sock, input, strlen(input), 0);

        data_len = recv(sock, output, MAX_DATA, 0);
        if (data_len <= 0) break;

        output[data_len] = '\0';
        printf("From server: %s\n", output);
    }

    close(sock);
    return 0;
}
